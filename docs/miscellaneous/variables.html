Default value : winston.format.combine( commonFormat, winston.format.colorize(), // Colorize part of the message winston.format.printf(({ level, message, timestamp }) => { return `${timestamp} ${level}: ${message}`; }) )

Default value : (): string => { const env = process.env.NODE_ENV || DEV_ENV; const isDevelopment = env === DEV_ENV; return isDevelopment ? 'debug' : 'warn'; }

Type : object
Default value : { error: 0, warn: 1, info: 2, http: 3, debug: 4 }
